// Level.js
class Level {
  constructor(id, tilemap) {
    this.id = id;
    this.tilemap = tilemap;
    this.entities = [];
    this.spawnPoint = null;
  }
  load(assets) {
    return assets.preload(this.tilemap)
      .then(map => this.parse(map));
  }
  spawn(entity) {
    entity.position = this.spawnPoint;
    this.entities.push(entity);
  }
}
